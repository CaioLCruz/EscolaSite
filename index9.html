                    int LDR = A0;

                      pinMode(LED, OUTPUT);
                      pinMode(LDR, INPUT);

                      int luz = analogRead(LDR);
                      Serial.println(luz);

                        digitalWrite(LED, HIGH);
                      } 
                      else {
                        digitalWrite(LED, LOW);
                      }
                    }
